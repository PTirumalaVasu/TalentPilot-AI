### JavaScript

```javascript
function getDemoStateParam() {
  return new URLSearchParams(window.location.search).get('demo_state');
}

function setGridState(state) {
  // state: 'loading' | 'loaded' | 'empty' | 'error'
  document.getElementById('grid-loading-state').classList.toggle('hidden', state !== 'loading');
  document.getElementById('grid-empty-state').classList.toggle('hidden', state !== 'empty');
  document.getElementById('grid-error-state').classList.toggle('hidden', state !== 'error');
  document.getElementById('skills-dashboard-grid-skills').classList.toggle('hidden', state !== 'loaded');
  document.getElementById('skills-dashboard-pagination-controls').classList.toggle('hidden', state !== 'loaded');
}

async function loadDashboard() {
  const forcedState = getDemoStateParam();

  if (forcedState === 'loading') {
    setGridState('loading');
    console.log('🔧 demo_state=loading — showing skeleton indefinitely');
    return;
  }
  if (forcedState === 'error') {
    setGridState('error');
    console.log('🔧 demo_state=error — showing error state');
    return;
  }

  setGridState('loading');
  const assignments = forcedState === 'empty' ? [] : await PrototypeAPI.getAssignments();
  console.log('📊 Assignments loaded:', assignments);

  if (assignments.length === 0) {
    setGridState('empty');
    document.getElementById('skills-dashboard-summary-count').textContent = 'Total: 0 assignments';
    return;
  }

  document.getElementById('skills-dashboard-summary-count').textContent =
    `Total: ${assignments.length} assignment${assignments.length === 1 ? '' : 's'}`;
  document.getElementById('assignments-tbody').innerHTML = assignments.map(rowHtml).join('');
  setGridState('loaded');
}
```

Note: `loadDashboard()` is being modified (was written in Section 2) rather than added new — this replaces the Section 2 version.

## Common Issues & Fixes

### Issue: Table flashes briefly before skeleton/empty/error shows
**Symptom**: Loaded table visible for a split second
**Cause**: `setGridState('loaded')` was the default HTML state before JS runs
**Fix**: Table and pagination should start with no `hidden` class removed by default in markup, since `setGridState()` runs synchronously on page load before paint in practice — verify no flash during testing.

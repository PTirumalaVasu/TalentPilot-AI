## Goal

Add the modal's Loading and Error states from the spec's Page States table. Extend the `?demo_state=` convention (from Section 3) with `modal-loading` / `modal-error` to auto-open the modal in that forced state on page load, since real data loading is too fast to naturally show these states.

### JavaScript

```javascript
let lastDrillDownId = null;

async function openDrillDown(assignmentId, forceState) {
  lastDrillDownId = assignmentId;
  document.getElementById('assignment-details-modal').classList.remove('hidden');

  if (forceState === 'loading') {
    setModalState('loading');
    console.log('🔧 demo_state=modal-loading — showing modal skeleton indefinitely');
    return;
  }
  if (forceState === 'error') {
    setModalState('error');
    console.log('🔧 demo_state=modal-error — showing modal error state');
    return;
  }

  setModalState('loading');
  const a = await PrototypeAPI.getAssignmentDetail(assignmentId);
  console.log('🔍 Drill-down opened for', a.employee.firstName, '-', a.skill.name);

  document.getElementById('assignment-details-header-id').textContent =
    `Employee: ${a.employee.firstName} ${a.employee.lastName} | Skill: ${a.skill.name}`;
  document.getElementById('assignment-details-provenance-summary').innerHTML = provenanceBadge(a);
  document.getElementById('assignment-details-raw-signal').innerHTML = renderRawSignal(a);
  document.getElementById('assignment-details-explanation').textContent = renderExplanation(a);

  setModalState('loaded');
}

function retryDrillDown() {
  if (lastDrillDownId) openDrillDown(lastDrillDownId);
}

function setModalState(state) {
  // state: 'loading' | 'loaded' | 'error'
  document.getElementById('assignment-details-loading').classList.toggle('hidden', state !== 'loading');
  document.getElementById('assignment-details-error').classList.toggle('hidden', state !== 'error');
  document.getElementById('assignment-details-provenance-summary').classList.toggle('hidden', state !== 'loaded');
  document.getElementById('assignment-details-raw-signal').classList.toggle('hidden', state !== 'loaded');
  document.getElementById('assignment-details-explanation').classList.toggle('hidden', state !== 'loaded');
}
```

## Common Issues & Fixes

### Issue: Header stays empty during modal-loading/modal-error forced states
**Symptom**: Header shows nothing since it's only set in the "loaded" branch
**Cause**: Spec says header ("Employee: X | Skill: Y") should stay visible even while the body loads — but forced-error/forced-loading skip the fetch entirely, so there's no employee/skill data to show yet
**Fix**: Acceptable for the demo-only forced states (header can stay blank or show a generic placeholder) since real loading/error in production would still have the row's employee/skill known synchronously before the async fetch — document this as a known limitation of the demo-only trigger, not a real spec violation.

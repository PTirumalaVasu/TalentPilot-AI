### JavaScript

```javascript
window.initPage = function() {
  console.log('📄 02.1 Content Discovery loaded');
  loadAssignment();
};

async function loadAssignment() {
  const a = await PrototypeAPI.getContentDiscovery();
  console.log('📦 Content Discovery data:', a);

  document.getElementById('card-skill-title').textContent = a.skillName;
  document.getElementById('card-status').textContent = `${a.status}`;
  document.getElementById('content-discovery-content-title').textContent = a.content.title;
  document.getElementById('content-discovery-source-badge').textContent = a.content.source;
  document.getElementById('content-discovery-duration').textContent = `${a.content.durationMinutes} minutes`;
  document.getElementById('content-discovery-description').textContent = a.content.description;
  document.getElementById('content-discovery-contact-info').textContent = `Questions? Contact Rita`;
  document.getElementById('content-discovery-progress-indicator').textContent = '0% watched';
}

function handlePlay() {
  console.log('▶ Play clicked — would launch video player at 0:00 (real player out of scope for this prototype)');
}

function handleAlternatives() {
  console.log('ℹ️ View alternatives clicked (browse UI not built — out of scope for this prototype)');
}
```

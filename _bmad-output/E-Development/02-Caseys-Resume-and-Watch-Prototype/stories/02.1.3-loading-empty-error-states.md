### JavaScript

```javascript
function getDemoStateParam() {
  return new URLSearchParams(window.location.search).get('demo_state');
}

function setCardState(state) {
  // state: 'loading' | 'loaded' | 'empty' | 'error'
  document.getElementById('card-loading-state').classList.toggle('hidden', state !== 'loading');
  document.getElementById('card-empty-state').classList.toggle('hidden', state !== 'empty');
  document.getElementById('card-error-state').classList.toggle('hidden', state !== 'error');
  document.querySelector('#page-main-content > .bg-white').classList.toggle('hidden', state !== 'loaded');
}

async function loadAssignment() {
  const forcedState = getDemoStateParam();

  if (forcedState === 'loading') { setCardState('loading'); return; }
  if (forcedState === 'error') { setCardState('error'); return; }
  if (forcedState === 'empty') { setCardState('empty'); return; }

  setCardState('loading');
  const a = await PrototypeAPI.getContentDiscovery();
  console.log('📦 Content Discovery data:', a);

  document.getElementById('card-skill-title').textContent = a.skillName;
  document.getElementById('card-status').textContent = a.status;
  document.getElementById('content-discovery-content-title').textContent = a.content.title;
  document.getElementById('content-discovery-source-badge').textContent = a.content.source;
  document.getElementById('content-discovery-duration').textContent = `${a.content.durationMinutes} minutes`;
  document.getElementById('content-discovery-description').textContent = a.content.description;
  document.getElementById('content-discovery-contact-info').textContent = 'Questions? Contact Rita';
  document.getElementById('content-discovery-progress-indicator').textContent = '0% watched';
  setCardState('loaded');
}
```

Note: `loadAssignment()` replaces Section 2's version (adds state branching) — not a new function.

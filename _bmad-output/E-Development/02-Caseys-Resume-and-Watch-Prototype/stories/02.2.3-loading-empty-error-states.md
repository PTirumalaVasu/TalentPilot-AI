### JavaScript

```javascript
function getDemoStateParam() {
  return new URLSearchParams(window.location.search).get('demo_state');
}

function setCardState(state) {
  document.getElementById('cw-loading-state').classList.toggle('hidden', state !== 'loading');
  document.getElementById('cw-empty-state').classList.toggle('hidden', state !== 'empty');
  document.getElementById('cw-error-state').classList.toggle('hidden', state !== 'error');
  document.querySelector('#page-main-content > .bg-white').classList.toggle('hidden', state !== 'loaded');
}

async function loadContinueWatching() {
  const forcedState = getDemoStateParam();

  if (forcedState === 'loading') { setCardState('loading'); return; }
  if (forcedState === 'error') { setCardState('error'); return; }
  if (forcedState === 'empty') { setCardState('empty'); return; }

  setCardState('loading');
  const c = await PrototypeAPI.getContinueWatching();
  console.log('📦 Continue Watching data:', c);

  document.getElementById('cw-skill-title').textContent = c.skillName;
  document.getElementById('cw-status').textContent = `${c.status} · ${c.lastUpdate}`;
  document.getElementById('continue-watching-content-title').textContent = c.content.title;
  document.getElementById('continue-watching-source-badge').textContent = c.content.source;
  document.getElementById('continue-watching-duration').textContent = `${c.content.durationMinutes} minutes`;
  document.getElementById('continue-watching-progress-text').textContent =
    `${c.watchedMinutes} min watched of ${c.totalMinutes} min · ${c.totalMinutes - c.watchedMinutes} min remaining`;
  document.getElementById('continue-watching-btn-resume').textContent = `▶ Resume at ${c.resumeTimestamp}`;
  document.getElementById('continue-watching-encouragement').textContent = `You're making progress on ${c.skillName}!`;
  setCardState('loaded');
}
```

Note: `loadContinueWatching()` replaces Section 2's version (adds state branching).

### JavaScript

```javascript
let assignmentFlowState = { employeeId: null, skillId: null, content: null };

async function openAssignmentFlow() {
  assignmentFlowState = { employeeId: null, skillId: null, content: null };
  document.getElementById('skill-assignment-flow-modal').classList.remove('hidden');
  goToStep(1);

  const employees = await PrototypeAPI.getEmployees();
  const dropdown = document.getElementById('skill-assignment-flow-employee-dropdown');
  dropdown.innerHTML = '<option value="">Select employee or search...</option>' +
    employees.map(e => `<option value="${e.id}">${escapeHtml(e.firstName + ' ' + e.lastName)}</option>`).join('');
}

function closeAssignmentFlow() {
  document.getElementById('skill-assignment-flow-modal').classList.add('hidden');
}

function onEmployeeSelected() {
  const val = document.getElementById('skill-assignment-flow-employee-dropdown').value;
  assignmentFlowState.employeeId = val || null;
  document.getElementById('skill-assignment-flow-btn-continue-step1').disabled = !val;
}

function goToStep(n) {
  document.getElementById('skill-assignment-flow-progress-indicator').textContent = `Step ${n} of 3`;
  // Section 2/3 will add step-2/step-3 show/hide logic here
}
```

Note: `openAssignmentFlow()` will be wired to the `[+ New Assignment]` button in Section 6 (replacing the current stub). For this section, it can be tested by calling `openAssignmentFlow()` manually from the console.

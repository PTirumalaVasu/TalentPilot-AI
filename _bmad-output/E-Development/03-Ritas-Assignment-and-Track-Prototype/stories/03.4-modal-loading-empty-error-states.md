## Goal

Add the 3 non-happy-path states from the retrofit, previewable via `?demo_state=asf-loading|asf-empty|asf-error`. Simplification for this prototype: one generic Error state represents any of the 3 possible list-fetch failures (employees/skills/content), rather than building 3 separately distinguishable error UIs — sufficient for UX validation, not exhaustive QA.

### JavaScript

```javascript
function getDemoStateParam() {
  return new URLSearchParams(window.location.search).get('demo_state');
}

function setAsfState(state) {
  // state: 'loading' | 'error' | 'normal'
  document.getElementById('asf-loading-state').classList.toggle('hidden', state !== 'loading');
  document.getElementById('asf-error-state').classList.toggle('hidden', state !== 'error');
  document.getElementById('asf-step-1').classList.toggle('hidden', state !== 'normal' || assignmentFlowState.currentStep !== 1);
}
```

**Update `openAssignmentFlow()`** to check for forced demo states:

```javascript
async function openAssignmentFlow() {
  assignmentFlowState = { employeeId: null, skillId: null, content: null, currentStep: 1 };
  document.getElementById('skill-assignment-flow-modal').classList.remove('hidden');

  const forced = getDemoStateParam();
  if (forced === 'asf-loading') { setAsfState('loading'); return; }
  if (forced === 'asf-error') { setAsfState('error'); return; }

  setAsfState('normal');
  goToStep(1);

  const employees = await PrototypeAPI.getEmployees();
  // ... rest unchanged
}
```

**Update `onSkillSelected()`** to show the Empty state when no content matches, or when `?demo_state=asf-empty` forces it:

```javascript
async function onSkillSelected() {
  const val = document.getElementById('skill-assignment-flow-skill-search').value;
  assignmentFlowState.skillId = val || null;
  document.getElementById('skill-assignment-flow-btn-continue-step2').disabled = !val;

  if (val) {
    const forced = getDemoStateParam();
    assignmentFlowState.content = (forced === 'asf-empty') ? null : await PrototypeAPI.getContentForSkill(val);
  }
}
```

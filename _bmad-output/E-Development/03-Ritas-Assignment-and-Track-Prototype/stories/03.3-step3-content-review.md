**Update `goToStep(n)`** to also show/hide Step 3 and populate content fields when entering Step 3:

```javascript
function goToStep(n) {
  document.getElementById('skill-assignment-flow-progress-indicator').textContent = `Step ${n} of 3`;
  document.getElementById('asf-step-1').classList.toggle('hidden', n !== 1);
  document.getElementById('asf-step-2').classList.toggle('hidden', n !== 2);
  document.getElementById('asf-step-3').classList.toggle('hidden', n !== 3);
  document.getElementById('skill-assignment-flow-btn-assign').disabled = (n !== 3);

  if (n === 3 && assignmentFlowState.content) {
    const c = assignmentFlowState.content;
    document.getElementById('skill-assignment-flow-content-title').textContent = c.title;
    document.getElementById('skill-assignment-flow-content-source').textContent = c.source;
    document.getElementById('skill-assignment-flow-content-duration').textContent = `${c.durationMinutes} minutes`;
    document.getElementById('skill-assignment-flow-content-description').textContent = c.description;
  }
}
```
